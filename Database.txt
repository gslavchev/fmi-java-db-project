CREATE TABLE CLIENTS (
   CLIENT_ID INT AUTO_INCREMENT PRIMARY KEY,
   CLIENT_NAME NVARCHAR(50) NOT NULL,
   CLIENT_AGE INT NOT NULL,
   CLIENT_GENDER NVARCHAR(4) NOT NULL,
   CLIENT_CITY NVARCHAR(20) NOT NULL
);
CREATE TABLE CARS (
    CAR_ID INT AUTO_INCREMENT PRIMARY KEY,
    CAR_MODEL NVARCHAR(80) NOT NULL,
    YEAR_OF_PRODUCTION INT NOT NULL,
    HORSE_POWER INT NOT NULL,
    GEARBOX NVARCHAR(40) NOT NULL,
    FUEL NVARCHAR(40) NOT NULL,
    MILEAGE INT NOT NULL,
    PRICE DECIMAL(14, 2),
    CONSTRAINT CH_PRICE CHECK (PRICE > 0)
);
CREATE TABLE SALES (
    SALES_ID INT AUTO_INCREMENT PRIMARY KEY,
    CAR_ID INT NOT NULL,
    CLIENT_ID INT NOT NULL,
    BUYING_DATE DATE,
    CONSTRAINT FK_CAR FOREIGN KEY (CAR_ID)
    REFERENCES CARS (CAR_ID),
    CONSTRAINT FK_CLIENTS FOREIGN KEY (CLIENT_ID)
    REFERENCES CLIENTS (CLIENT_ID)
);
